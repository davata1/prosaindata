import requests
from bs4 import BeautifulSoup
import pandas as pd
import string
import nltk
from nltk.corpus import stopwords
from nltk.tokenize import word_tokenize
from nltk.stem import PorterStemmer
import random

nltk.download('punkt')
nltk.download('stopwords')

# Fungsi untuk menghapus tanda baca
def remove_punctuation(text):
    return text.translate(str.maketrans('', '', string.punctuation))

# Fungsi untuk menghapus stop words
stop_words = set(stopwords.words('english'))
def remove_stopwords(text):
    words = word_tokenize(text)
    filtered_words = [word for word in words if word.lower() not in stop_words]
    return ' '.join(filtered_words)

# Fungsi untuk melakukan stemming
stemmer = PorterStemmer()
def stem_text(text):
    words = word_tokenize(text)
    stemmed_words = [stemmer.stem(word) for word in words]
    return ' '.join(stemmed_words)

# Fungsi untuk normalisasi teks abstrak
def normalize_abstract(text):
    text = remove_punctuation(text)
    text = remove_stopwords(text)
    text = stem_text(text)
    text = text.lower()
    text = text.strip()

    return text

# Fungsi untuk mengubah teks abstrak menjadi Term Frequency
def abstract_to_tf(text):
    # Tokenisasi teks
    tokens = word_tokenize(text)

    # Jika abstrak kosong atau tidak ada kata, kembalikan nilai default 0
    if len(tokens) == 0:
        return {'N/A': 0}

    # Hitung frekuensi kemunculan setiap kata dalam teks
    word_count = {}
    for word in tokens:
        if word in word_count:
            word_count[word] += 1
        else:
            word_count[word] = 1

    # Hitung TF untuk setiap kata
    tf = {}
    total_words = len(tokens)
    for word, count in word_count.items():
        tf[word] = count / total_words

    return tf

# Assuming you have a column 'Abstrak' containing abstract text

    # Jika abstrak kosong atau tidak ada kata, kembalikan nilai default 0
    if len(tokens) == 0:
        return {'N/A': 0}

    # Hitung frekuensi kemunculan setiap kata dalam teks
    word_count = {}
    for word in tokens:
        if word in word_count:
            word_count[word] += 1
        else:
            word_count[word] = 1

    # Hitung TF untuk setiap kata
    tf = {}
    total_words = len(tokens)
    for word, count in word_count.items():
        tf[word] = count / total_words

    return tf

# Mengganti nilai-nilai dalam TF menjadi Binary Frequency (1 jika lebih dari 0, 0 jika 0)
df['Abstrak_Binary_Frequency'] = df['Abstrak_TF'].apply(lambda x: [1 if val > 0 else 0 for val in x.values()])

# Menampilkan DataFrame dengan kolom Binary Frequency
